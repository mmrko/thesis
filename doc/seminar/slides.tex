  \item \textbf{fotoluminesenssi}: valon fotoni virittää elektronin ylemmälle viritystilalle, jonka purkautuessa atomi emittoi valoa tietyllä aallonpituusalueella (värivaikutelma)

\end{itemize}

\begin{frame}{Toteutus}

\begin{itemize}
  \setlength\itemsep{1em}
  \item luminoforin virittäminen ulkoiseen \textbf{kameramoduuliin} integroidun valolähteen (Yongnuo YN565EX) avulla
  \item emission (värimuutoksen) kaappaaminen: \textbf{n} kappaletta kuvia \textbf{m} millisekunnin välein minimitarkennusetäisyydeltä
  \item YCbCr-RGB-HSV-muunnos $\Rightarrow$ dominoivat värisävyt (hue) $\Rightarrow$ sormenjälki (matriisi)
  \item Sormenjälkien vertailu värisävyn arvon ja intensiteetin perusteella (mukautettu Hungarian-algoritmi)
\end{itemize}

\end{frame}

\begin{frame}{Tulokset ja tutkimushaasteet}
  \begin{itemize}
    \setlength\itemsep{1em}
    \item 16 luminoforia $\times$ 9 kuvausprofiilia = 144 uniikkia \textbf{tägiä}
    \item Kaksi eri kuvadatan vertailumetodia: sormenjälki- sekä histogrammipohjainen menetelmä
    \item Pitempi kaappausaikaväli parantaa tarkkuutta
    \item Luminoforien valmistusprosessilla suuri vaikutus: sammumisnopeus, konsentraatioerot, viritysaallonpituus
    \item Haasteita: viritysvalon ja luminoforin yhteensovittaminen, luminoforien syntentisointi ja kestävyys (e.g. photobleach)
  \end{itemize}
\end{frame}
